@RestController
@RequestMapping(value = "/analysisReport")
public class AnalysisReportController implements IAnalysisReportController {
    private final EntLogger logger = EntLogFactory.getSanitizedLogger(this.getClass());

    /**
     * COMPONENT EXISTENCE ANALYSIS REPORT
     */
    @RestAccessControl(permission = Permission.SUPERUSER)
    @PostMapping(value = "/component/diff", produces = MediaType.APPLICATION_JSON_VALUE)

    public ResponseEntity<SimpleRestResponse<AnalysisReportResponse>> runComponentExistenceAnalysisReport(
            @RequestBody Map<String, List<String>> idCodesByObjectType) {
        //-
        Set<String> objectTypes = idCodesByObjectType.keySet();
        logger.debug("Running runAnalysisReport on types {}", objectTypes);
        return runComponentExistenceAnalysisReportDefaultImpl(
                new ComponentExistenceAnalysis(),
                idCodesByObjectType);
    }

    @Override
    public Object mapComponentTypeToService(String objectType) {
        switch (AnalysisReportResponse.ObjectType.valueOf(objectType)) {
            case widgets:
                return widgetService;
            case fragments:
                return guiFragmentService;
            case pages:
                return pageService;
            case pageTemplates:
                return pageModelService;
            case categories:
                return categoryService;
            case groups:
                return groupService;
            case labels:
                return labelService;
            case languages:
                return languageService;
            case resources:
                return fileBrowserService;
            default:
                MapBindingResult bindingResult = new MapBindingResult(new HashMap<>(), "analysisReport.Component.Diff");
                bindingResult.rejectValue("objectType", "1", new String[]{objectType}, "generic.notValid");
                throw new ValidationGenericException(bindingResult);
        }
    }

    @Autowired
    IWidgetService widgetService;
    @Autowired
    IGuiFragmentService guiFragmentService;
    @Autowired
    IPageService pageService;
    @Autowired
    IPageModelService pageModelService;
    @Autowired
    ICategoryService categoryService;
    @Autowired
    IGroupService groupService;
    @Autowired
    ILabelService labelService;
    @Autowired
    ILanguageService languageService;
    @Autowired
    IFileBrowserService fileBrowserService;
}
