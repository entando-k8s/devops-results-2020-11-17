package org.entando.entando.web.analysisreport;

import com.fasterxml.jackson.annotation.JsonValue;

import java.util.Map;

public class AnalysisReportResponse {

    @JsonValue
    private final Map<String, Map<String, String>> report;

    AnalysisReportResponse(Map<String, Map<String, String>> report) {
        this.report = report;
    }

    public enum Status {
        NEW("NEW"),
        DIFF("DIFF"),
        EQUAL("EQUAL");

        public final String value;

        Status(String value) {
            this.value = value;
        }
    }

    public enum ObjectType {
        widgets, fragments, pages, pageTemplates,           //NOSONAR
        categories, groups, labels, languages, resources,   //NOSONAR
    }
}
