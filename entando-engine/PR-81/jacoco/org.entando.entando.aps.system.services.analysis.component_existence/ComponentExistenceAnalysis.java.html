package org.entando.entando.aps.system.services.analysis.component_existence;

import org.entando.entando.aps.system.services.category.ICategoryService;
import org.entando.entando.aps.system.services.group.IGroupService;
import org.entando.entando.aps.system.services.guifragment.IGuiFragmentService;
import org.entando.entando.aps.system.services.label.ILabelService;
import org.entando.entando.aps.system.services.language.ILanguageService;
import org.entando.entando.aps.system.services.page.IPageService;
import org.entando.entando.aps.system.services.page.PageService;
import org.entando.entando.aps.system.services.pagemodel.IPageModelService;
import org.entando.entando.aps.system.services.storage.IFileBrowserService;
import org.entando.entando.aps.system.services.widgettype.IWidgetService;
import org.entando.entando.ent.exception.EntRuntimeException;

import java.util.List;

public class ComponentExistenceAnalysis implements IComponentExistenceAnalysis {

    private boolean doesComponentExistsOnService(Object service, String idCode) {
        try {
            if (service instanceof IWidgetService) {
                return ((IWidgetService) service).exists(idCode);
            } else if (service instanceof IGuiFragmentService) {
                return ((IGuiFragmentService) service).exist(idCode);
            } else if (service instanceof PageService) {
                return ((IPageService) service).exists(idCode, IPageService.STATUS_DRAFT) ||
                        ((IPageService) service).exists(idCode, IPageService.STATUS_ONLINE);
            } else if (service instanceof IPageModelService) {
                return ((IPageModelService) service).exists(idCode);
            } else if (service instanceof ICategoryService) {
                return ((ICategoryService) service).exists(idCode);
            } else if (service instanceof IGroupService) {
                return ((IGroupService) service).exists(idCode);
            } else if (service instanceof ILabelService) {
                return ((ILabelService) service).exists(idCode);
            } else if (service instanceof ILanguageService) {
                return ((ILanguageService) service).exists(idCode);
            } else if (service instanceof IFileBrowserService) {
                return ((IFileBrowserService) service).exists(idCode, false);
            } else if (service == null) {
                throw new EntRuntimeException("Null service type detected");
            } else {
                throw new EntRuntimeException("Illegal service type detected: \"" + service.getClass().getSimpleName() + "\"");
            }
        } catch (Exception ex) {
            throw new EntRuntimeException("Error detected checking component existence" + idCode, ex);
        }
    }

    @Override
    public ComponentExistenceAnalysisResult run(ComponentExistenceAnalysisResult result, List<ServiceParams> servicesParams) {
        //-
        for (ServiceParams serviceParams : servicesParams) {
            Object service = serviceParams.service;
            List<String> idCodesToCheck = serviceParams.idCodesList;

            ComponentStatusMap componentStatusMap = result.map.getOrDefault(service, new ComponentStatusMap());

            for (String idCodeToCheck : idCodesToCheck) {
                if (doesComponentExistsOnService(service, idCodeToCheck)) {
                    componentStatusMap.map.put(idCodeToCheck, Status.DIFF);
                } else {
                    componentStatusMap.map.put(idCodeToCheck, Status.NEW);
                }
            }

            result.map.put(service, componentStatusMap);
        }
        return result;
    }
}
