package org.entando.entando.web.common.model;

import org.entando.entando.ent.exception.EntException;

import javax.validation.constraints.NotNull;

/**
 * Class to support both common and named ids:
 * <pre>
 * A named id is an key-value assignment expression in the form:
 * - {name}={value}
 * please note that "name" can be a zero-length string, this way
 * you will still be allowed to enter no-name identifiers containing equals signs
 * A common id is just an identified that doesn't contain the separator
 * </pre>
 */
public class RestNamedId {

    public static final char SEPARATOR = '=';
    private static final String NO_NAME = "";
    public final String name;
    public final String value;

    public RestNamedId(String rawId) {
        int pos = rawId.indexOf(SEPARATOR);
        if (pos == -1) {
            name = NO_NAME;
            value = rawId;
        } else {
            name = rawId.substring(0, pos);
            value = rawId.substring(pos + 1);
        }
    }

    /**
     * Name validation gate.
     * Throws exception if object name != from the provided name
     *
     * @return itself, unless fails with an exception
     * @throws EntException if the validation fails
     */
    public RestNamedId assertName(@NotNull String name) throws EntException {
        if (checkName(name)) {
            return this;
        } else {
            throw new EntException(String.format("Invalid Id Name detected, expected: \"%s\" found: \"%s\""
                    , this.name
                    , name
            ));
        }
    }

    /**
     * Name validation check.
     * Tells if the object name equals the one provided
     */
    public boolean checkName(@NotNull String name) {
        return (name != null && this.name.equals(name));
    }

    /**
     * Build an object form a raw id
     */
    public static RestNamedId from(String rawId) {
        return new RestNamedId(rawId);
    }
}
